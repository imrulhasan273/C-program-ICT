## Switch example 

```c
#include <stdio.h>
int main() 
{
    int n;
    
    
    printf("Enter number between 1 to 3: ");
    
    scanf("%d", &n);
    
    switch(n)
    {
        case 1:
            printf("Book1");
            break;
        case 2:
            printf("Book2");
            break;
        case 3:
            printf("Book 3");
            break;
        default:
            printf("Invalid Number");
    }
    
    return 0;
}
```

> Never forget to use `break` after every case.

# **Star Pattern**


```c
#include <stdio.h>

int main()
{
    int n;

    scanf("%d", &n);
    
    int i;
    int j;
    
    for(i=1 ; i<=n ; i++)
    {
       for(j=1;j<=i;j++)
       {
           printf("*");
       }
       printf("\n");
    }
    

    return 0;
}
```

### Output

```
*
**
***
****
*****
```

# While Loop

## 1+2+....+n

```c
#include <stdio.h>
int main()
{
    
    int n;
    scanf("%d", &n);
    int sum=0;
    int i=1;
    while(i<=n)
    {
        sum = sum+i;
        i++;
    }
    
    printf("%d", sum);
      
    return 0;
}
```
